ax = plt.subplot(projection='polar')
ax.spines[:].set_visible(False)
# ax.spines[:].set_color('red')
ax.set_yticks([])
x = np.array([0, 1, 2, 3, 4, 5, 6]) / 3 * np.pi
y = [93, 91, 88, 93, 74, 85, 93]
labels = ['语文', '数学', '英语', '物理', '化学', '生物']
labels = [label + '\n' + str(score) for label, score in zip(labels, y)] + ['']
ax.set_xticks(x)

ax.set_ylim((0, 100))
ax.plot(x, [100] * 7, c='lightgrey')
ax.plot(x, [66] * 7, c='lightgrey')
ax.plot(x, [33] * 7, c='lightgrey')
for i in x:
    ax.axvline(i, c='lightgrey')
    
# ax.tick_params(axis='y', color='r')
ax.plot(x, y)
ax.fill(x, y, alpha=.4)
ax.set_xticklabels(labels, fontsize=20)
ax.tick_params(pad=12)
plt.show()
