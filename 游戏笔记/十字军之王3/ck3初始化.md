CK3 DNA 文本解析器

```python
class CK3TxtParser:
    def __init__(self):
        self.pos = 0

    def parse(self, json_string):
        json_string = json_string.strip('\n \t')
        self.pos = 0
        return self.parse_object(json_string)

    def parse_value(self, json_string):
        start = self.pos
        while len(json_string) > self.pos:
            a = json_string[self.pos]
            if a == '{':
                self.pos += 1
                v = self.parse_object(json_string)
                return v
            elif a in [" ", "\n", "\t"]:
                if self.pos != start:
                    value = json_string[start: self.pos]
                    self.pos += 1
                    return value
                self.pos += 1
                self.start = self.pos
            else:
                self.pos += 1
        
        return json_string[start: self.pos-1]
    

    def parse_object(self, json_string):
        result = {}
        while (len(json_string) > self.pos) and (json_string[self.pos] != "}"):
            key = self.parse_string(json_string)
            
            # print("key:", key)
            if type(key) is list:
                return key
            # 一定是找到等于号了
            if key is not None:
                value = self.parse_value(json_string)
                result[key] = value
        self.pos += 1
        return result
    
    def parse_string(self, json_string):
        start = self.pos
        array = []
        while json_string[self.pos] != "}":
            a = json_string[self.pos]
            
            if a in [" ", "\n", "\t"]:
                if self.pos != start:
                    array.append(json_string[start: self.pos])
                self.pos += 1
                start = self.pos
                continue
                
            if a == "=":
                if len(array) == 0:
                    array.append(json_string[start: self.pos])
                key = array[0]
                self.pos += 1
                return key
            
            self.pos += 1
        
        v = json_string[start: self.pos]
        if len(v) > 0:
            array.append(v)
        
        
        if len(array) > 0:
            self.pos += 1
            return [int(i) if i.isdigit() else i for i in array]
```

```python
def to_string(dna):
    res = ""
    for k, v in dna.items():
        res += k
        res += " = "
        if type(v) is list:
            res += "{ "
            for i in  v:
                res += f"{str(i)} "
            res += "}"
        elif type(v) is dict:
            res += "{ "
            res += to_string(v)
            res += "}"
        elif type(v) is str:
            res += v
        else:
            print(type(v))
            raise ValueError()
        res += "\n"
    return res
```
